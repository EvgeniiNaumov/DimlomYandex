Задание 1:
SELECT
    "courierId",
    COUNT(*) AS in_delivery_count
FROM "Orders"
WHERE "inDelivery" = true
GROUP BY "courierId"
ORDER BY "courierId";


Задание 2:
SELECT
    track,
    CASE
        WHEN finished = true THEN 2
        WHEN cancelled = true THEN -1
        WHEN "inDelivery" = true THEN 1
        ELSE 0
    END AS status
FROM "Orders"
ORDER BY track;
